from collections import Counter

IS_PART_1 = True

def fetch_input() -> tuple[list[int], list[int]]:
    left, right = [], []

    with open("input.txt") as f:
        lines = f.readlines()
    
    for line in lines:
        a, b = line.split('  ')
        left.append(int(a))
        right.append(int(b.strip()))
    
    left.sort()
    right.sort()

    return left, right

def part_1(left: list[int], right: list[int]) -> int:
    total = 0
    for l, r in zip(left, right):
        total += abs(l - r)

    return total

def part_2(left: list[int], right: list[int]) -> int:
    c = Counter(right)
    total = 0

    for l in left:
        total += l * c[l]

    return total

def main():
    inputs = fetch_input()
    f = part_1 if IS_PART_1 else part_2

    print(f(*inputs))

if __name__ == "__main__":
    main()
